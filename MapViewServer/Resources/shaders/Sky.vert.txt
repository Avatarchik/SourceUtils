attribute vec3 aPosition;

varying vec3 vWorldPos;

uniform mat4 uProjection;
uniform mat4 uView;
uniform mat4 uModel;

void main()
{
    vWorldPos = (uModel * vec4(aPosition, 1.0)).xyz;
    vec4 pos = uProjection * uView * vec4(vWorldPos.xyz, 1.0);

    pos.z = pos.w - 1.0 / 256.0;

    gl_Position = pos;
}
